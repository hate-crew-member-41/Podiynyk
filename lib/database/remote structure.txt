"Firebase Firestore" is used for the remote database (see firebase.google.com/products/firestore).
Below is the structure of the database.


collection 'HEIs' {
	document 'index' {
		$id: String HEI,
		...
	}

	document $HEI-id {
		'departments': <String, String>{
			$id: String department,
			...
		}
	},
	...
},

collection 'groups' {
	document $HEI-id.$department-id.$group {
		'leader': bool,
		'joined': DateTime
	},
	...
},

collection 'students' {
	document $group-id {
		$student: String role,
		...
	},
	...
},

collection 'confirmations' {
	ephemeral document '$group-id' {
		$student: int confirmations,
		...
	},
	...
},

collection 'subjects' {
	document $group-id {
		$subject-id: <String, dynamic>{
			'name': String,
			'now': int,
			'total': int
		},
		...

		collection $subject-id {
			document 'events' {
				$id: <String, dynamic>{
					'title': String,
					'datetime': DateTime,
					'note'?: String
				},
				...
			},
			document 'info' {
				$id: String content
			}
		},
		...
	},
	...
}
